git branch - посмотреть название ветки
в разных ветках можно делать разные модули - но лучше делать в одной ветке чтобы потом ничего не сливать
git branch название  - создать ветку с данным названием

git status - посмотреть список измененных файлов в данной ветке
git diff имя файла - посмтреть изменения в файле(смотреть лучше через прогу)
git add . - добавить все свои изменения
git add название файла - добавить в список измененных только файл
git commit -m "название_коммита" - закоммитить изменения

git stash - спрятать изменения
git stash list - посмотреть список спрятанных изменений
git stash apply - вернуть спрятанные изменения

git reset --hard HEAD откатывает изменения к указанному коммиту

git revert sha_коммита - сделать коммит который отменяет данный

git pull - слить изменения с сервера
git pull origin название_ветки - слить изменения с сервера по конкретноф ветке
git pull --rebase - данная опция показывает что при сливе изменений твои изменения ставятся в верхушку


git push origin вылить на сервер в ветку мастер
git push origin название_ветки - вылить изменения в конкретную ветку
